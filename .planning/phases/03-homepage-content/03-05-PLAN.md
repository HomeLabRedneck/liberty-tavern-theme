<!-- GJS connector minimal pattern (Discourse 3.x, Glimmer) -->
// A connector with access to the parent component's @category arg:
import Component from "@glimmer/component";

export default class TavernRoomPreview extends Component {
  // this.args.category is injected by Discourse's connector system
  // Properties available on a Discourse Category model:
  //   category.latestTopicTitle   — string or null (may not exist on all installs)
  //   category.topicUrl           — string "/c/slug/id" (always present)
  //   category.name               — string (always present)
  //   category.color              — hex string without # (e.g., "90302E")
  //   category.url                — string (same as topicUrl)

  get latestTitle() {
    return this.args.category?.latestTopicTitle ?? null;
  }

  get categoryUrl() {
    return this.args.category?.url ?? `/c/${this.args.category?.slug}`;
  }

  get categoryColor() {
    const c = this.args.category?.color;
    return c ? `#${c}` : "var(--tertiary)";
  }

  <template>
    {{#if this.latestTitle}}
      <div class="tavern-room-preview">
        <a href={{this.categoryUrl}} class="tavern-room-preview__link">
          <span class="tavern-room-preview__chevron">▶</span>
          <span class="tavern-room-preview__title">{{this.latestTitle}}</span>
        </a>
      </div>
    {{/if}}
  </template>
}

<!-- Fallback if latestTopicTitle is not available -->
// When latestTitle is null (property not exposed), render a "View latest →" link:
  get latestTitle() {
    return this.args.category?.latestTopicTitle
      ?? this.args.category?.latest_topic_title
      ?? null;
  }
// If null: render nothing ({{#if}} guard hides element — card looks like 03-04 result)

<!-- From tavern-banner.gjs — established import pattern -->

```javascript
// Connector outlet: {outlet-name-chosen-in-task-1}
// Renders a latest-topic preview line inside each room card.
// @category is injected by Discourse's connector system from the category-boxes component.
// If category.latestTopicTitle is not available on this Discourse install,
// the {{#if}} guard suppresses the element and the card degrades to 03-04 state.
import Component from "@glimmer/component";

export default class TavernRoomPreview extends Component {
  get latestTitle() {
    // Try both camelCase and snake_case — Discourse serializer varies by version
    return (
      this.args.category?.latestTopicTitle ??
      this.args.category?.latest_topic_title ??
      null
    );
  }

  get categoryUrl() {
    return (
      this.args.category?.url ??
      `/c/${this.args.category?.slug ?? ""}/${this.args.category?.id ?? ""}`
    );
  }

  <template>
    {{#if this.latestTitle}}
      <div class="tavern-room-preview">
        <a href={{this.categoryUrl}} class="tavern-room-preview__link">
          <span class="tavern-room-preview__chevron" aria-hidden="true">▶</span>
          <span class="tavern-room-preview__title">{{this.latestTitle}}</span>
        </a>
      </div>
    {{/if}}
  </template>
}
```
